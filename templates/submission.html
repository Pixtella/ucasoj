                        #include <stdio.h>
#include <string.h>
char s[205],t[205];
int n,m,a[205],b[205];
int main(){
    scanf("%s%s",s,t);
    n=strlen(s),m=strlen(t);
    for (int i=1;i<=n;++i) a[i]=s[n-i]-'0';
    for (int i=1;i<=m;++i) b[i]=t[m-i]-'0';
    if (m>n) n=m;
    for (int i=1;i<=n;++i) a[i+1]+=(a[i]+=b[i])/10,a[i]%=10;
    if (a[n+1]) ++n;
    for (int i=n;i;--i) putchar(a[i]+'0');
    puts("");
    return 0;
}
